:javaOpts
REM If JAVA_OPTS specified, use those options
REM Otherwise, default Java to using 256MB of memory
if "%JAVA_OPTS%"=="" set JAVA_OPTS=-Xmx256m


Стало:
:javaOpts
REM If JAVA_OPTS specified, use those options
REM Otherwise, default Java to using 256MB of memory
::if "%JAVA_OPTS%"=="" set JAVA_OPTS=-Xmx256m
set JAVA_OPTS=-Xms1536m -Xmx1536m -XX:MaxPermSize=768m
